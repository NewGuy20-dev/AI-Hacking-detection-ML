│   ├── unsw_nb15/        # UNSW-NB15 (8 files)

├── fraud_detection/       # 144 MB

│   └── domains/          # Alexa Top 1M
│
└── email_spam/            # 15 MB
    └── spam_corpus       # Spam email corpus
